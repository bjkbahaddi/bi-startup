// 1. البيانات الافتراضية const defaultData = [ { name: "تيك نوفا", category: "technology", icon: "fa-microchip", desc: "تطوير أنظمة الذكاء الاصطناعي المتقدمة." }, { name: "سريع باي", category: "fintech", icon: "fa-wallet", desc: "حلول الدفع الإلكتروني الفوري للمتاجر." }, { name: "دكتور أونلاين", category: "healthcare", icon: "fa-user-nurse", desc: "استشارات طبية مرئية على مدار الساعة." }, { name: "متجري", category: "ecommerce", icon: "fa-shopping-bag", desc: "أنشئ متجرك الإلكتروني في أقل من دقيقة." } ]; // 2. جلب البيانات من الذاكرة المحلية let startups = JSON.parse(localStorage.getItem('myStartups')); // 3. التحقق من وجود بيانات if (!startups || startups.length === 0) { startups = defaultData; localStorage.setItem('myStartups', JSON.stringify(defaultData)); } // 4. دالة عرض الشركات في الـ Grid function render(items) { const grid = document.getElementById('startupGrid'); if (!grid) return; if (items.length === 0) { grid.innerHTML = '

لا توجد شركات في هذه الفئة حالياً.

'; return; } grid.innerHTML = items.map(item => `

${item.name}

${item.desc}

${translateCategory(item.category)}
`).join(''); } // دالة ترجمة الفئات function translateCategory(cat) { const cats = { 'technology': 'تكنولوجيا', 'fintech': 'مالية', 'healthcare': 'صحة', 'ecommerce': 'تجارة' }; return cats[cat] || cat; } // 5. دالة الفلترة (تعمل عند الضغط على الأزرار) window.filterData = function(cat, btn) { // تحديث شكل الأزرار document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active')); if(btn) btn.classList.add('active'); // تصفية البيانات const filtered = (cat === 'all') ? startups : startups.filter(d => d.category === cat); render(filtered); } // 6. تشغيل العرض الأولي عند تحميل الصفحة document.addEventListener('DOMContentLoaded', () => { render(startups); // ربط نموذج التواصل const form = document.getElementById('form'); if(form) { form.addEventListener('submit', (e) => { e.preventDefault(); alert('شكراً لرسالتك! تم استلام طلبك بنجاح.'); form.reset(); }); } });
